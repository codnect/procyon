Lightweight application framework for Go

Build clean, production-ready Go applications.

Procyon provides dependency injection, configuration, lifecycle management, profiles, and web support — while keeping the simplicity and performance of Go.

Get Started GitHub 

package main

import (
    "os"

    "codnect.io/procyon"
)

func main() {
    app := procyon.New()

    if err := app.Run(os.Args...); err != nil {
        os.Exit(1)
    }
}
